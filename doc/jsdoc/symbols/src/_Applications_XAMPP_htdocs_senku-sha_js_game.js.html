 72 			}

 77 /**
 78  * Adds all event handlers for the game
 79  */
 80 
 81 Game.prototype.addHandlers = function() {
 82 
 83 	$(document).keydown(function (e) {
 84 			
 85 		if (game && game.ball && game.track) {
 86 		
 87 			switch (e.which) {      
 88 
 89 				case CONFIG.KEYCODE.LEFT:
 90 					if (game.ball.canMove.left) {
 91 						
 92 						game.track.speedX = CONFIG.ACCELERATION*game.controlDirection;
 93 					}
 94 					break;
 95 													 
 96 				case CONFIG.KEYCODE.RIGHT:
 97 					if (game.ball.canMove.right) {
 98 						
 99 						game.track.speedX = -1*CONFIG.ACCELERATION*game.controlDirection;  
100 					}
101 					break;
102 											  
103 				case CONFIG.KEYCODE.UP:                
104 					if (!game.isFinished) {
105 						
106 						game.track.start();
107 					}
108 					break;
109 
110 				case CONFIG.KEYCODE.SPACE:
111 					game.ball.jump();
112 					break;
113 			}
114 		}
115 	});  
116 
117 	$(document).keyup(function (e) {
118 											
119 		if (game && game.ball && game.track) {
120 		
121 			switch (e.keyCode) {      
122 
123 				case CONFIG.KEYCODE.LEFT:			 
124 				case CONFIG.KEYCODE.RIGHT:
125 					game.track.speedX = 0;
126 					break;
127 			}
128 		}
129 	});  
130 
131 	$(window).resize(function() {
132 							 
133 		if (game && game.renderer) {
134 		
135 			game.container.css({
136 				
137 				width: $(window).innerWidth(),
138 				height: $(window).innerHeight(),
139 				background: "url(img/game_background.jpg)"
140 			});
141 			game.camera.aspect = game.container.width()/game.container.height();
142 			game.camera.updateProjectionMatrix();
143 			game.renderer.setSize(game.container.width(), game.container.height());
144 		}
145 	});
146 	
147 	$(window).mousemove(function(e) {
148 		
149 		game.mousePos = [e.clientX, e.clientY];
150 	});
151 	
152 	$(window).click(function(e) {
153 		
154 		game.isMouseControlled = true;
155 		if (game.track.isStarted) {
156 			
157 			game.ball.jump();
158 		}
159 		else {
160 			
161 			game.track.start();
162 		}
163 	});
164 };

170 Game.prototype.startGame = function() {
171 	
172 	this.isFinished = false;
173 	this.controlDirection = 1;
174 	this.warpEndTime = 0;
175 	this.trackPosition = new THREE.Vector3(0, 0, 0);
176 
177 	this.timeDifference = 0;
178 	this.lastFrameTime = new Date();
179 	
180 	this.scene = new THREE.Scene();
181 	this.container = $("#content");
182 	this.container.html("<div id=\"levelName\" />");
183 	this.container.css({
184 		
185 		width: $(window).innerWidth(),
186 		height: $(window).innerHeight(),
187 		background: "url(img/game_background.jpg)"
188 	});
189 	
190 	this.pointLight = new THREE.PointLight(0xffffff);
191 	this.pointLight.position.set(0,5,7);     
192 	this.scene.add(this.pointLight);     
193 
194 	this.spotLight = new THREE.SpotLight(0xffffff);
195 	this.spotLight.position.set(0,5,0);
196 	this.spotLight.shadowCameraNear = 0.01;		
197 	this.spotLight.castShadow = true;
198 	this.spotLight.shadowDarkness = 0.5;  
199 	this.scene.add(this.spotLight);
200 
201 	this.renderer = new THREE.WebGLRenderer();
202 	this.renderer.setSize(this.container.width(), this.container.height());
203 	this.renderer.shadowMapEnabled = true;
204 	this.renderer.shadowMapSoft = true;
205 	this.renderer.setClearColor( CONFIG.BACKGROUND_COLOR, 0 );     
206 	
207 	this.scene.fog = new THREE.Fog(CONFIG.BACKGROUND_COLOR, CONFIG.TRACK.FOG_NEAR, CONFIG.TRACK.FOG_FAR);
208 
209 	this.track = new Track(this.currentLevel, this.isTest);
210 	this.track.nextX = this.track.position.x;
211 	this.scene.add(this.track); 
212 	
213 	this.ball = new Ball();
214 	this.scene.add(this.ball);          
215 
216 	this.camera = new THREE.PerspectiveCamera(60, this.container.width()/this.container.height(), .01, 1000);
217 	this.camera.position.y = 2;
218 	this.camera.position.z = 5;
219 	this.camera.lookAt(this.ball.position);       
220 
221    	this.container.get(0).appendChild(this.renderer.domElement);
222 	
223 	this.render();
224 
225 	this.addHandlers();
226 };

228 /**
229  * The rendering method called on each frame
230  * @param {Number} time Current timestamp, used to synchronize game speed
231  */
232 
233 Game.prototype.render = function(time) {
234 
235 	this.renderProcess = requestAnimationFrame(this.render.bind(this));     
236 													
237 	this.timeDifference = time-this.lastFrameTime;
238 	this.lastFrameTime = time;				  
239 	this.timeRate = this.timeDifference / 200;
240 	
241 	if (this.track && this.ball && !this.track.isStopped) {
242 	
243 		if (this.isInGoal && !this.track.isStarted) {
244 			
245 			this.track.position = this.track.nextGoalPosition();
246 		}
247 		else {
248 			
249 			if (!this.isInGoal) {
250 				
251 				this.levelTime = new Date();
252 				this.levelTime -= this.startTime;
253 				this.levelTime = Math.floor(this.levelTime/1000);
254 				Util.updateInfoHTML();
255 			}
256 			this.track.position = this.track.nextPosition();
257 		}
258 		this.track.updateBlocks();
259 		this.ball.rotateAroundWorldAxis(new THREE.Vector3(1,0,0), -1*this.track.tempSpeedZ);
260 		this.ball.rotateAroundWorldAxis(new THREE.Vector3(0,0,1), this.track.tempSpeedX);  
261 	}
262 						 
263 	this.renderer.render(this.scene, this.camera);
264 };

266 /**
267  * Clears the game and returns to menu or editor, without parameters, the menu will be opened.
268  * @param {Number} editLevelId ID of the level, if set, editor will be opened
269  * @param {Boolean} edit If true, the level will be loaded in editor, otherwise, level selection will be opened 
270  */
271 
272 Game.prototype.clearGame = function(editLevelId, edit) {
273 
274 	cancelAnimationFrame(this.renderProcess);
275 	$(window).unbind();
276 	$("*").unbind();
277 	this.container.css({
278 		
279 		background: "none"
280 	})
281 	bgBall = new BackgroundBall();
282 	$("#info").css({
283 		display: "none"
284 	});	
285 	if (editLevelId) {
286 		
287 		var param = "";
288 		if (edit) {
289 			
290 			param = "?id=" + this.currentLevel;
291 		}
292 		Util.changeContent("editor.php" + param);
293 	}
294 	else {
295 		
296 		Util.changeContent("menu.php");
297 	}
298 	this.currentLevel = 1;
299 	delete this;
300 };
