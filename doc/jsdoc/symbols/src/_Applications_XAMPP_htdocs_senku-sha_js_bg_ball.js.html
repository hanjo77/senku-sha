  7 function BackgroundBall() {
  8 
  9 	this.renderProcess;
 10 	this.renderer;
 11 	this.container;
 12 	this.camera;
 13 	this.scene;
 14 	this.ball;
 15 	this.speedX;
 16 	this.speedY;
 17 	this.speedZ;

 25 
 26 /**
 27  * Start the background ball
 28  */
 29 
 30 BackgroundBall.prototype.start = function () {
 31 
 32     this.speedX = .01;
 33     this.speedY = .01;
 34     this.speedZ = .01;
 35 
 36     this.scene = new THREE.Scene();
 37     this.container = $("#bgBall");
 38 
 39     this.pointLight = new THREE.PointLight(0xffffff);
 40     this.pointLight.position.set(0, 5, 7);
 41     this.scene.add(this.pointLight);
 42 
 43     this.spotLight = new THREE.SpotLight(0xffffff);
 44     this.spotLight.position.set(0, 5, 0);
 45     this.spotLight.shadowCameraNear = 0.01;
 46     this.spotLight.castShadow = true;
 47     this.spotLight.shadowDarkness = 0.5;
 48     this.scene.add(this.spotLight);
 49 
 50     this.renderer = new THREE.WebGLRenderer();
 51     this.renderer.setSize(this.container.width(), this.container.height());
 52     this.renderer.shadowMapEnabled = true;
 53     this.renderer.shadowMapSoft = true;
 54     this.renderer.setClearColor(CONFIG.BACKGROUND_COLOR, 0);
 55 
 56     this.scene.fog = new THREE.Fog(CONFIG.BACKGROUND_COLOR, CONFIG.TRACK.FOG_NEAR, CONFIG.TRACK.FOG_FAR);
 57 
 58     this.ball = new Ball();
 59     this.scene.add(this.ball);
 60 
 61     this.camera = new THREE.PerspectiveCamera(45, this.container.width() / this.container.height(), 1, 1000);
 62     this.camera.position.z = 1.5;
 63     this.camera.position.y = 1;
 64     this.camera.lookAt(this.ball.position);
 65 
 66     this.container.get(0).appendChild(this.renderer.domElement);
 67 
 68     this.render();
 69 };
 70 
 71 /**
 72  * The rendering method called on each frame
 73  */
 74 
 75 BackgroundBall.prototype.render = function() {
 76 		
 77 	this.renderProcess = requestAnimationFrame(this.render.bind(this));   
 78 									
 79 	if (mouseX && mouseY) {
 80 		
 81 		this.speedX = (mouseX-($(document).width()/2))/10000;
 82 		this.speedZ = (mouseY-($(document).height()/2))/10000;
 83 	}
 84 	
 85 	this.ball.rotateAroundWorldAxis(new THREE.Vector3(1,0,0), this.speedZ);
 86 	this.ball.rotateAroundWorldAxis(new THREE.Vector3(0,1,0), this.speedX);
 87 																			
 88 	this.renderer.render(this.scene, this.camera);
 89 };
