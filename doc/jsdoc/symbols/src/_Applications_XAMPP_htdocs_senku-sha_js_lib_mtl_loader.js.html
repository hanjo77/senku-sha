1 /**
  2  * Loads a Wavefront .mtl file specifying materials
  3  *
  4  * @author angelxuanchang
  5  */
  6 
  7 THREE.MTLLoader = function( baseUrl, options, crossOrigin ) {
  8 
  9 	this.baseUrl = baseUrl;
 10 	this.options = options;
 11 	this.crossOrigin = crossOrigin;
 12 
 13 };
 14 
 15 THREE.MTLLoader.prototype = {
 16 
 17 	constructor: THREE.MTLLoader,
 18 
 19 	/**
 20 	 * Loads a MTL file
 21 	 *
 22 	 * Loading progress is indicated by the following events:
 23 	 *   "load" event (successful loading): type = 'load', content = THREE.MTLLoader.MaterialCreator
 24 	 *   "error" event (error loading): type = 'load', message
 25 	 *   "progress" event (progress loading): type = 'progress', loaded, total
 26 	 *
 27 	 * @param url - location of MTL file
 28 	 */
 29 	load: function( url ) {
 30 
 31 		var scope = this;
 32 		var xhr = new XMLHttpRequest();
 33 
 34 		function onloaded( event ) {
 35 
 36 			if ( event.target.status === 200 || event.target.status === 0 ) {
 37 
 38 				var materialCreator = scope.parse( event.target.responseText );
 39 
 40 				// Notify caller, that I'm done
 41 
 42 				scope.dispatchEvent( { type: 'load', content: materialCreator } );
 43 
 44 			} else {
 45 
 46 				scope.dispatchEvent( { type: 'error', message: 'Couldn\'t load URL [' + url + ']',
 47 					response: event.target.responseText } );
 48 
 49 			}
 50 
 51 		}
 52 
 53 		xhr.addEventListener( 'load', onloaded, false );
 54 
 55 		xhr.addEventListener( 'progress', function ( event ) {
 56 
 57 			scope.dispatchEvent( { type: 'progress', loaded: event.loaded, total: event.total } );
 58 
 59 		}, false );
 60 
 61 		xhr.addEventListener( 'error', function () {
 62 
 63 			scope.dispatchEvent( { type: 'error', message: 'Couldn\'t load URL [' + url + ']' } );
 64 
 65 		}, false );
 66 
 67 		xhr.open( 'GET', url, true );
 68 		xhr.send( null );
 69 	},
 70 
 71 	/**
 72 	 * Parses loaded MTL file
 73 	 * @param text - Content of MTL file
 74 	 * @return {THREE.MTLLoader.MaterialCreator}
 75 	 */
 76 	parse: function( text ) {
 77 
 78 		var lines = text.split( "\n" );
 79 		var info = {};
 80 		var delimiter_pattern = /\s+/;
 81 		var materialsInfo = {};
 82 
 83 			for ( var i = 0; i < lines.length; i ++ ) {
 84 
 85 			var line = lines[ i ];
 86 			line = line.trim();
 87 
 88 			if ( line.length === 0 || line.charAt( 0 ) === '#' ) {
 89 
 90 				// Blank line or comment ignore
 91 				continue;
 92 
 93 			}
 94 
 95 			var pos = line.indexOf( ' ' );
 96 
 97 			var key = ( pos >= 0 ) ? line.substring( 0, pos) : line;
 98 			key = key.toLowerCase();
 99 
100 			var value = ( pos >= 0 ) ? line.substring( pos + 1 ) : "";
101 			value = value.trim();
102 
103 			if ( key === "newmtl" ) {
104 
105 				// New material
106 
107 				info = { name: value };
108 				materialsInfo[ value ] = info;
109 
110 			} else if ( info ) {
111 
112 				if ( key === "ka" || key === "kd" || key === "ks" ) {
113 
114 					var ss = value.split( delimiter_pattern, 3 );
115 					info[ key ] = [ parseFloat( ss[0] ), parseFloat( ss[1] ), parseFloat( ss[2] ) ];
116 
117 				} else {
118 
119 					info[ key ] = value;
120 
121 				}
122 
123 			}
124 
125 		}
126 
127 		var materialCreator = new THREE.MTLLoader.MaterialCreator( this.baseUrl, this.options );
128 		materialCreator.setMaterials( materialsInfo );
129 		return materialCreator;
130 
131 	}
132 
133 };

135 /**
136  * Create a new THREE-MTLLoader.MaterialCreator
137  * @param baseUrl - Url relative to which textures are loaded
138  * @param options - Set of options on how to construct the materials
139  *                  side: Which side to apply the material
140  *                        THREE.FrontSide (default), THREE.BackSide, THREE.DoubleSide
141  *                  wrap: What type of wrapping to apply for textures
142  *                        THREE.RepeatWrapping (default), THREE.ClampToEdgeWrapping, THREE.MirroredRepeatWrapping
143  *                  normalizeRGB: RGBs need to be normalized to 0-1 from 0-255
144  *                                Default: false, assumed to be already normalized
145  *                  ignoreZeroRGBs: Ignore values of RGBs (Ka,Kd,Ks) that are all 0's
146  *                                  Default: false
147  *                  invertTransparency: If transparency need to be inverted (inversion is needed if d = 0 is fully opaque)
148  *                                      Default: false (d = 1 is fully opaque)
149  * @constructor
150  */
151 
152 THREE.MTLLoader.MaterialCreator = function( baseUrl, options ) {
153 
154 	this.baseUrl = baseUrl;
155 	this.options = options;
156 	this.materialsInfo = {};
157 	this.materials = {};
158 	this.materialsArray = [];
159 	this.nameLookup = {};
160 
161 	this.side = ( this.options && this.options.side )? this.options.side: THREE.FrontSide;
162 	this.wrap = ( this.options && this.options.wrap )? this.options.wrap: THREE.RepeatWrapping;
163 
164 };
165 
166 THREE.MTLLoader.MaterialCreator.prototype = {
167 
168 	constructor: THREE.MTLLoader.MaterialCreator,
169 
170 	setMaterials: function( materialsInfo ) {
171 
172 		this.materialsInfo = this.convert( materialsInfo );
173 		this.materials = {};
174 		this.materialsArray = [];
175 		this.nameLookup = {};
176 
177 	},
178 
179 	convert: function( materialsInfo ) {
180 
181 		if ( !this.options ) return materialsInfo;
182 
183 		var converted = {};
184 
185 		for ( var mn in materialsInfo ) {
186 
187 			// Convert materials info into normalized form based on options
188 
189 			var mat = materialsInfo[ mn ];
190 
191 			var covmat = {};
192 
193 			converted[ mn ] = covmat;
194 
195 			for ( var prop in mat ) {
196 
197 				var save = true;
198 				var value = mat[ prop ];
199 				var lprop = prop.toLowerCase();
200 
201 				switch ( lprop ) {
202 
203 					case 'kd':
204 					case 'ka':
205 					case 'ks':
206 
207 						// Diffuse color (color under white light) using RGB values
208 
209 						if ( this.options && this.options.normalizeRGB ) {
210 
211 							value = [ value[ 0 ] / 255, value[ 1 ] / 255, value[ 2 ] / 255 ];
212 
213 						}
214 
215 						if ( this.options && this.options.ignoreZeroRGBs ) {
216 
217 							if ( value[ 0 ] === 0 && value[ 1 ] === 0 && value[ 1 ] === 0 ) {
218 
219 								// ignore
220 
221 								save = false;
222 
223 							}
224 						}
225 
226 						break;
227 
228 					case 'd':
229 
230 						// According to MTL format (http://paulbourke.net/dataformats/mtl/):
231 						//   d is dissolve for current material
232 						//   factor of 1.0 is fully opaque, a factor of 0 is fully dissolved (completely transparent)
233 
234 						if ( this.options && this.options.invertTransparency ) {
235 
236 							value = 1 - value;
237 
238 						}
239 
240 						break;
241 
242 					default:
243 
244 						break;
245 				}
246 
247 				if ( save ) {
248 
249 					covmat[ lprop ] = value;
250 
251 				}
252 
253 			}
254 
255 		}
256 
257 		return converted;
258 
259 	},
260 
261 	preload: function () {
262 
263 		for ( var mn in this.materialsInfo ) {
264 
265 			this.create( mn );
266 
267 		}
268 
269 	},
270 
271 	getIndex: function( materialName ) {
272 
273 		return this.nameLookup[ materialName ];
274 
275 	},
276 
277 	getAsArray: function() {
278 
279 		var index = 0;
280 
281 		for ( var mn in this.materialsInfo ) {
282 
283 			this.materialsArray[ index ] = this.create( mn );
284 			this.nameLookup[ mn ] = index;
285 			index ++;
286 
287 		}
288 
289 		return this.materialsArray;
290 
291 	},
292 
293 	create: function ( materialName ) {
294 
295 		if ( this.materials[ materialName ] === undefined ) {
296 
297 			this.createMaterial_( materialName );
298 
299 		}
300 
301 		return this.materials[ materialName ];
302 
303 	},
304 
305 	createMaterial_: function ( materialName ) {
306 
307 		// Create material
308 
309 		var mat = this.materialsInfo[ materialName ];
310 		var params = {
311 
312 			name: materialName,
313 			side: this.side
314 
315 		};
316 
317 		for ( var prop in mat ) {
318 
319 			var value = mat[ prop ];
320 
321 			switch ( prop.toLowerCase() ) {
322 
323 				// Ns is material specular exponent
324 
325 				case 'kd':
326 
327 					// Diffuse color (color under white light) using RGB values
328 
329 					params[ 'diffuse' ] = new THREE.Color().setRGB( value[0], value[1], value[2] );
330 
331 					break;
332 
333 				case 'ka':
334 
335 					// Ambient color (color under shadow) using RGB values
336 
337 					params[ 'ambient' ] = new THREE.Color().setRGB( value[0], value[1], value[2] );
338 
339 					break;
340 
341 				case 'ks':
342 
343 					// Specular color (color when light is reflected from shiny surface) using RGB values
344 					params[ 'specular' ] = new THREE.Color().setRGB( value[0], value[1], value[2] );
345 
346 					break;
347 
348 				case 'map_kd':
349 
350 					// Diffuse texture map
351 
352 					params[ 'map' ] = this.loadTexture( this.baseUrl + value );
353 					params[ 'map' ].wrapS = this.wrap;
354 					params[ 'map' ].wrapT = this.wrap;
355 
356 					break;
357 
358 				case 'ns':
359 
360 					// The specular exponent (defines the focus of the specular highlight)
361 					// A high exponent results in a tight, concentrated highlight. Ns values normally range from 0 to 1000.
362 
363 					params['shininess'] = value;
364 
365 					break;
366 
367 				case 'd':
368 
369 					// According to MTL format (http://paulbourke.net/dataformats/mtl/):
370 					//   d is dissolve for current material
371 					//   factor of 1.0 is fully opaque, a factor of 0 is fully dissolved (completely transparent)
372 
373 					if ( value < 1 ) {
374 
375 						params['transparent'] = true;
376 						params['opacity'] = value;
377 
378 					}
379 
380 					break;
381 
382 				default:
383 					break;
384 
385 			}
386 
387 		}
388 
389 		if ( params[ 'diffuse' ] ) {
390 
391 			if ( !params[ 'ambient' ]) params[ 'ambient' ] = params[ 'diffuse' ];
392 			params[ 'color' ] = params[ 'diffuse' ];
393 
394 		}
395 
396 		this.materials[ materialName ] = new THREE.MeshPhongMaterial( params );
397 		return this.materials[ materialName ];
398 
399 	},
400 
401 
402 	loadTexture: function ( url, mapping, onLoad, onError ) {
403 
404 		var isCompressed = /\.dds$/i.test( url );
405 
406 		if ( isCompressed ) {
407 
408 			var texture = THREE.ImageUtils.loadCompressedTexture( url, mapping, onLoad, onError );
409 
410 		} else {
411 
412 			var image = new Image();
413 			var texture = new THREE.Texture( image, mapping );
414 
415 			var loader = new THREE.ImageLoader();
416 			loader.crossOrigin = this.crossOrigin;
417 			loader.load( url, function ( image ) {
418 
419 				texture.image = THREE.MTLLoader.ensurePowerOfTwo_( image );
420 				texture.needsUpdate = true;
421 
422 				if ( onLoad ) onLoad( texture );
423 
424 			} );
425 
426 		}
427 
428 		return texture;
429 
430 	}
431

434 THREE.MTLLoader.ensurePowerOfTwo_ = function ( image ) {
435 
436 	if ( ! THREE.MTLLoader.isPowerOfTwo_( image.width ) || ! THREE.MTLLoader.isPowerOfTwo_( image.height ) ) {
437 
438 		var canvas = document.createElement( "canvas" );
439 		canvas.width = THREE.MTLLoader.nextHighestPowerOfTwo_( image.width );
440 		canvas.height = THREE.MTLLoader.nextHighestPowerOfTwo_( image.height );
441 
442 		var ctx = canvas.getContext("2d");
443 		ctx.drawImage( image, 0, 0, image.width, image.height, 0, 0, canvas.width, canvas.height );
444 		return canvas;
445 
446 	}
447 
448 	return image;
449 
450 };

452 THREE.MTLLoader.isPowerOfTwo_ = function ( x ) {
453 
454 	return ( x & ( x - 1 ) ) === 0;
455 
456 };
457 
458 THREE.MTLLoader.nextHighestPowerOfTwo_ = function( x ) {
459 
460 	--x;
461 
462 	for ( var i = 1; i < 32; i <<= 1 ) {
463 
464 		x = x | x >> i;
465 
466 	}
467 
468 	return x + 1;
469 
470 };
