  9 function Ball() {
 10 	       
 11 	// call parent constructor
 12 	THREE.Mesh.call(this);
 13 	this.geometry = new THREE.SphereGeometry(CONFIG.BALL_RADIUS, 24, 16);  
 14 	this.position.y = CONFIG.BALL_RADIUS;
 15 	
 16 	this.texture = THREE.ImageUtils.loadTexture('img/ball_texture.png');
 17 	this.material = new THREE.MeshLambertMaterial({
 18 		map: this.texture,                 
 19 	}),                                                                     
 20 	                             
 21 	this.castShadow = true;
 22 	this.receiveShadow = false;
 23 	this.lastBlock = null;
 24 	this.isJumping = false;
 25 	this.isFalling = true;
 26 	this.inAir = false;
 27 	this.canMove = {
 28 		
 29 		back: true,
 30 		front: true,
 31 		left: true,
 32 		right: true
 33 	};
 34 }
 35 
 36 // inherit THREE.Mesh
 37 Ball.prototype = new THREE.Mesh();
 38 Ball.prototype.constructor = Ball;
 39 
 40 /**
 41  * The point directly under the ball
 42  * @returns Vector3 object of the ball's bottom touch point
 43  * @type THREE.Vector3
 44  */
 45 
 46 Ball.prototype.touchPoint = function() {
 47 	
 48 	return new THREE.Vector3(  
 49 		
 50 		this.position.x + this.geometry.radius,
 51 		this.position.y,
 52 		this.position.z + this.geometry.radius
 53 	);
 54 }
 55 
 56 /**
 57  * The point that will be under the ball in the next frame
 58  * @returns Vector3 object of the ball's next bottom touch point
 59  * @type THREE.Vector3
 60  */
 61 
 62 Ball.prototype.nextTouchPoint = function() {
 63 	
 64 	return new THREE.Vector3(
 65 		                                                          
 66 		this.touchPoint().x + speedX,
 67 		this.touchPoint().y,
 68 		this.touchPoint().z + speedZ
 69 	);
 70 }

 72 /**
 73  * Rotate ball around an arbitrary axis in world space
 74  * @param {THREE.Vector3} axis The axis
 75  * @param {Number} radians The angle in radians
 76  */
 77       
 78 Ball.prototype.rotateAroundWorldAxis = function(axis, radians) {
 79 	
 80     var rotWorldMatrix = new THREE.Matrix4();
 81     rotWorldMatrix.makeRotationAxis(axis.normalize(), radians);
 82     rotWorldMatrix.multiply(this.matrix);        // pre-multiply
 83     this.matrix = rotWorldMatrix;
 84     this.rotation.setFromRotationMatrix(this.matrix);
 85 }

 87 Ball.prototype.jump = function() {
 88 	
 89 	if (!this.inAir) {
 90 
 91 		this.isJumping = true;
 92 	}
 93 }

 95 /**
 96  * Finds block under the ball
 97  * @param {THREE.Vector3} trackPosition Position of the track
 98  * @returns Block under ball
 99  * @type Block
100  */
101 
102 Ball.prototype.blockUnderBall = function(trackPosition) {
103 
104 	var pos = this.touchPoint();              
105 	var posOnTrack = new THREE.Vector3( 
106 		(-trackPosition.x+(CONFIG.BLOCK_SIZE/2)),
107 		-trackPosition.y,
108 		(-trackPosition.z+(CONFIG.BLOCK_SIZE/2))
109 	);
110 	if (posOnTrack.x >= 0 && posOnTrack.y >= 0) {
111 		
112 		var block = game.track.blockForPosition(
113 			parseInt(posOnTrack.x/CONFIG.BLOCK_SIZE, 10), 
114 			parseInt(posOnTrack.z/CONFIG.BLOCK_SIZE, 10)
115 		);
116 		if (block) {
117 			
118 			if (game && trackPosition.y <= 0 && !this.inAir && !game.isInGoal) {
119 				
120 				var now = new Date().getTime();					
121 				switch (block.blockType.name) {
122 				
123 					case "jumper":
124 					
125 						if (game.warpEndTime <= 0) {
126 							
127 							this.isJumping = true;
128 						}
129 						break;
130 				
131 					case "slowdown":
132 					
133 						if (game.track.speedModifier >= 1 && game.warpEndTime <= 0) {
134 						
135 							game.slowdownEndTime = now + 5000;
136 							game.track.speedModifier /= 1.5;
137 							window.clearTimeout(game.slowdownTimer);
138 							game.slowdownTimer = window.setTimeout(function() {
139 						
140 								if (game.warpEndTime <= 0) {
141 								
142 									game.track.speedModifier *= 1.5;
143 									game.slowdownEndTime = 0;	
144 								}
145 							}, 5000);
146 						}
147 						break;
148 				
149 					case "speedup":
150 					
151 						if (game.track.speedModifier <= 1 && game.warpEndTime <= 0) {
152 						
153 							game.speedupEndTime = now + 5000;
154 							game.track.speedModifier *= 1.5;
155 							window.clearTimeout(game.speedupTimer);
156 							game.speedupTimer = window.setTimeout(function() {
157 						
158 								if (game.warpEndTime <= 0) {
159 								
160 									game.track.speedModifier /= 1.5;
161 									game.speedupEndTime = 0;	
162 								}
163 							}, 5000);
164 						}
165 						break;
166 				
167 					case "invertor":
168 					
169 						if (game.controlDirection == 1 && game.warpEndTime <= 0) {
170 						
171 							game.invertorEndTime = now + 5000;
172 							game.controlDirection = -1;
173 							window.clearTimeout(game.invertorTimer);
174 							game.invertorTimer = window.setTimeout(function() {
175 						
176 								game.controlDirection = 1;
177 								game.invertorEndTime = 0;	
178 							}, 5000);
179 						}
180 						break;
181 				
182 					case "warp":
183 					
184 						game.warpEndTime = now + 5000;
185 						game.speedupEndTime = 0;
186 						game.slowdownEndTime = 0;
187 						game.track.speedModifier = 2;
188 						window.clearTimeout(game.warpTimer);
189 						game.warpTimer = window.setTimeout(function() {
190 					
191 							game.track.speedModifier = 1;
192 							game.warpEndTime = 0;	
193 						}, 5000);
194 						break;
195 				
196 					case "goalBright":
197 					case "goalDark":
198 					
199 						if (game.startTime > 0) {
200 							
201 							game.track.finishLevel();
202 						}
203 						break;
204 				}
205 			}
206 			else if (game && game.isInGoal) {
207 				
208 				if (block.blockType.name.indexOf("goal") == -1) {
209 					
210 					game.startTime = 0;
211 					game.isInGoal = false;
212 					game.startTime = new Date();
213 				}
214 			}		
215 			
216 			this.lastBlock = block;
217 			return block;
218 		}                  
219 	} 
220 	return null;
221 }
