 11 /**
 12  * Removes all children of a ThreeJS object
 13  * @param {THREE.Object3D} obj ThreeJS object
 14  */
 15 
 16 Util.removeChilds = function(obj) {
 17 
 18 	for (var child in obj.children) {
 19 	
 20 		Util.removeChilds(obj.children[child]);
 21 	}
 22 	
 23 	if (obj.geometry) {
 24 		
 25 		obj.geometry.dispose();
 26 	}                        
 27 	
 28 	if (obj.material) {
 29 		
 30 		obj.material.dispose();
 31 	}                        
 32 	
 33 	if (obj.texture) {
 34 		
 35 		obj.texture.dispose();
 36 	}                        
 37 	
 38 	if (obj.dispose) {
 39 	
 40 		obj.dispose();
 41 	}
 42 }

 70 /**
 71  * Returns the hex value of an integer to be used for a CSS color attribute
 72  * @param {String} intColor Integer value of color
 73  * @returns Hexadecimal value to be used for CSS color attribute
 74  * @type String
 75  */
 76 
 77 Util.getHexColorFromInt = function(intColor) {
 78 	
 79 	var color = intColor.toString(16);
 80 	while (color.length < 6) {
 81 	
 82 		color = "0" + color;
 83 	}
 84 	return "#" + color;

234 /**
235  * Changes the contents of the content-container and handles 
236  * the background-animation according to the content
237  * @param {String} page URL to replace the current contents
238  */
239 
240 
241 Util.changeContent = function(page) {
242 	
243 	if (page == "game.php") {
244 		
245 		$("#bgBall").html("");
246 		if (bgBall) {
247 		
248 			cancelAnimationFrame(bgBall.renderProcess);
249 		}
250 	}
251 	$('#content').html("");
252 	$.ajax({ url: page }).done(function( data ) {
253                        
254 		var id = 0;
255 		var contentDiv = $('#content');
256 		contentDiv.html(data);
257 		contentDiv.center();
258 		if (page.indexOf("id=") > -1) {
259 		
260 			id = page.substring(page.indexOf("id=") + 3);
261 		}
262 		if (page.indexOf("editor.php") > -1) {
263 			               
264 			editor = new Editor(id);
265 		}
266 		
267 		if (page != "game.php") {
268 			
269 			Util.initHandlers();  
270 		}
271 				
272 		var hash = "#";
273 		var content = page.substring(0, page.indexOf("."));
274 		if (content != "menu") {
275 			
276 			hash = "#" + page.substring(0, page.indexOf("."));
277 		}
278 		if (id > 0) {
279 		
280 			hash += "_" + id;
281 		}
282 		location.href = location.href.substring(0, location.href.indexOf("#")) + hash;
283 		Validation.formIsValid($('form').get(0));
284 		Util.updateWindow();		
285 	});
286 }

288 /**
289  * Reads the current URL hash and changes the content depending on it
290  */
291 
292 Util.handleHash = function() {
293 	
294 	var hash = "intro";
295 	var param = "";
296 	var hashPos = window.location.href.indexOf("#")+1;
297 	if (hashPos > 0) {
298 
299 		var hash = window.location.href.substring(hashPos);
300 		if (hash.indexOf("_") > -1) {
301 		
302 			var id = hash.substring(hash.indexOf("_") + 1);
303 			param = "?id=" + id;
304 			hash = hash.substring(0, hash.indexOf("_"));
305 		}
306 		else if (hash == "" || hash == "menu") {
307 
308 			hash = "menu";
309 			bgBall = new BackgroundBall();
310 		}
311 	} 
312 	Util.changeContent(hash + ".php" + param);
313 }

315 /**
316  * Returns the number of seconds until a given time.
317  * @param {Number} endTime Time value (Date.getTime())
318  * @returns Seconds until the time
319  * @type Number
320  */
321 
322 Util.getSecondsUntil = function(endTime) {
323 	
324 	if (endTime) {
325 		
326 		var difference = endTime-new Date().getTime();
327 		if (difference >= 0) {
328 			
329 			return Math.ceil(difference/1000);
330 		}
331 	}
332 	return null;
333 }

621 /**
622  * Returns an object of the collisions between the ball and a given block
623  * @param {Block} block BLOCK object to be tested
624  * @param {THREE.Vector3} nextPos THREE.Vector3 object of the next track position
625  * @returns String Array with the detected collisions (right|left|front|back)
626  * @type Array
627  * TODO: Make it better... ;-)
628  */
629 
630 Util.getCollisions = function(block, nextPos) {
631 	    
632 	var types = [];                              
633 	var ballPos = Util.getBallPosition(nextPos); 
634 	var movesLeft = (game.track.speedX > 0);
635 	var movesRight = (game.track.speedX < 0);
636 	var movesForward = (game.track.speedZ > 0);
637 	var movesBack = (game.track.speedZ < 0);
638 	var withinWidth = (ballPos.x > block.left) && (ballPos.x < block.right);
639 	var withinLength = (ballPos.z > block.front - (CONFIG.BLOCK_SIZE/2)) && (ballPos.z < block.back - (CONFIG.BLOCK_SIZE/2));
640 	var neighbours = block.neighbours;
641 
642 	var frontIntersection = ((ballPos.z <= block.front + (CONFIG.BLOCK_SIZE/2) + game.ball.geometry.radius)
643 		&& (ballPos.z > block.front)
644 		&& withinWidth);
645 	var backIntersection = ((ballPos.z >= (block.back + (CONFIG.BLOCK_SIZE/2)) - game.ball.geometry.radius)
646 		&& (ballPos.z < block.back)
647 		&& withinWidth);
648 	var leftIntersection = ((ballPos.x >= block.right - game.ball.geometry.radius) 
649 		&& (ballPos.x < block.left + (CONFIG.BLOCK_SIZE/2))
650 		&& withinLength);
651 	var rightIntersection = ((ballPos.x <= block.left + game.ball.geometry.radius ) 
652 		&& (ballPos.x > block.right - (CONFIG.BLOCK_SIZE/2))
653 		&& withinLength); 
654 	if (leftIntersection) {
655 
656 		types.push("left");
657 	}
658 	else if (rightIntersection) {
659 
660 		types.push("right");
661 	}
662 	if (frontIntersection) {
663                       
664 		types.push("front");
665 	}
666 	else if (backIntersection) {
667                                                       
668 		types.push("back");
669 	}      
670 	return types;
671 }

673 /**
674  * Centers a jQuery object relative to the window
675  */
676 
677 jQuery.fn.center = function () {
678     this.css({
679     	
680 		position: "absolute",
681 		height: "auto",
682 		top: Math.max(0, (($(window).height() - $(this).outerHeight()) / 2) + 
683                                                 $(window).scrollTop()) + "px",
684 		left: Math.max(0, (($(window).width() - $(this).outerWidth()) / 2) + 
685                                                 $(window).scrollLeft()) + "px"
686 	});
687     return this;
