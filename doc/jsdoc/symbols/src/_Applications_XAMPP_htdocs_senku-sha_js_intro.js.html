1 /**
  2  * The intro screen
  3  * @author Hanjo
  4  * @version $Rev$
  5  * @requires OtherClassName
  6  * @constructor
  7  */
  8 
  9 function Intro() {
 10 
 11 	this.renderProcess;
 12 	this.renderer;
 13 	this.container;
 14 	this.camera;
 15 	this.scene;
 16 	this.logo;
 17 	this.speedX;
 18 	this.speedY;
 19 	this.speedZ;
 20 	this.spotLight;
 21 	this.pointLight;
 22 	this.mouseMoveX;
 23 	this.mouseMoveY;
 24 	
 25 	this.start();
 26 }
 27 
 28 /**
 29  * Exits the intro screen
 30  */
 31 
 32 Intro.prototype.leave = function() {
 33 	
 34 	cancelAnimationFrame(this.renderProcess);
 35 	if (this.scene) {
 36 		
 37 		Util.removeChilds(this.scene);
 38 	}
 39 	Util.changeContent("menu.php");
 40 	bgBall = new BackgroundBall();
 41 }
 42 
 43 /**
 44  * Starts the intro screen
 45  */
 46 
 47 Intro.prototype.start = function() {
 48 
 49 	window.setTimeout(function() {
 50 
 51 		intro.leave();
 52 	}.bind(this), 5000);
 53 
 54 	this.speedX = .1;
 55 	this.speedY = .1;
 56 	this.speedZ = .1;                                                    
 57 
 58 	this.scene = new THREE.Scene();
 59 	this.container = $("#logo");
 60 	this.container.css({
 61 
 62 		width: $(window).innerWidth(),
 63 		height: $(window).innerHeight() 
 64 	});
 65 	var overlay = $("#introOverlay");
 66 	overlay.css({
 67 		
 68 		left: ($(window).innerWidth()-overlay.innerWidth())/2,
 69 		top: ($(window).innerHeight()-overlay.innerHeight())/2
 70 	})
 71 
 72 	this.pointLight = new THREE.PointLight(0xffffff);
 73 	this.pointLight.position.set(0,5,7);     
 74 	this.scene.add(this.pointLight);
 75 
 76 	this.renderer = new THREE.WebGLRenderer();
 77 	this.renderer.setSize(this.container.width(), this.container.height());
 78 	this.renderer.shadowMapEnabled = true;
 79 	this.renderer.shadowMapSoft = true;
 80 	this.renderer.setClearColor( 0x000000, 0 );     
 81 
 82 	this.camera = new THREE.PerspectiveCamera(45, this.container.width()/this.container.height(), 1, 1000); 
 83 	this.camera.position.z = 10;
 84 	this.camera.position.y = 0;
 85 
 86 	// model
 87 
 88 	var loader = new THREE.OBJMTLLoader();
 89 	loader.addEventListener( 'load', function ( event ) {
 90 
 91 		intro.logo = event.content;
 92 		intro.logo.position = {
 93 			x: 0,
 94 			y: 0,
 95 			z: 0
 96 		};
 97 		intro.camera.lookAt(intro.logo.position);       
 98 		intro.scene.add( intro.logo );
 99 
100 	});
101 	loader.load( 'obj/hanjo-logo.obj', 'obj/hanjo-logo.mtl' );
102 
103    	this.container.get(0).appendChild(this.renderer.domElement);
104 
105 	this.render();
106 }
107 
108 /**
109  * The rendering method called on each frame
110  */
111 
112 Intro.prototype.render = function() {
113 
114 	this.renderProcess = requestAnimationFrame(this.render.bind(this));   
115 					  
116 	if (this.logo) {
117 		
118 		this.logo.rotation.y += this.speedX; 
119 	}
120 																	
121 	this.renderer.render(this.scene, this.camera);
122 }
