  7 function Block(type, pos) {
  8 	       
  9 	// call parent constructor
 10 	THREE.Mesh.call(this);
 11 	this.blockHeight = (type.height ? type.height : CONFIG.BLOCK_HEIGHT)
 12 	this.geometry = new THREE.CubeGeometry(CONFIG.BLOCK_SIZE,this.blockHeight,CONFIG.BLOCK_SIZE);
 13 	this.position = new THREE.Vector3(           
 14 		(pos.x ? pos.x : 0),
 15 		-1*CONFIG.BLOCK_HEIGHT/2,
 16 		(pos.z ? pos.z : 0)
 17 	);                                        
 18 	this.material = new THREE.MeshLambertMaterial({color: (type.color ? type.color : 0x000000)});      
 19 	this.castShadow		= false;
 20 	this.receiveShadow	= true;
 21 	this.blockType = type;                
 22     this.left = this.position.x-(CONFIG.BLOCK_SIZE/2);
 23 	this.right = this.left+CONFIG.BLOCK_SIZE;
 24 	this.front = this.position.z;
 25 	this.back = this.front+CONFIG.BLOCK_SIZE;
 26 	this.neighbours = {};
 27 }

 38 Block.prototype.updateNeighbours = function() {
 39                                                       
 40 	var col = parseInt(this.position.x/CONFIG.BLOCK_SIZE, 10);
 41 	var row = parseInt(this.position.z/CONFIG.BLOCK_SIZE, 10);
 42 	this.neighbours = {                                    
 43  	    frontLeft: game.track.blockForPosition(col-1, row-1),
 44 		frontRight: game.track.blockForPosition(col+1, row-1),
 45 		backLeft: game.track.blockForPosition(col-1, row+1),
 46 		backRight: game.track.blockForPosition(col+1, row+1),
 47     	front: game.track.blockForPosition(col, row-1),
 48 		left: game.track.blockForPosition(col-1, row),
 49 		right: game.track.blockForPosition(col+1, row),
 50 		back: game.track.blockForPosition(col, row+1)
 51 	};
 52 }
