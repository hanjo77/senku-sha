  7 function Block(type, pos) {
  8 	       
  9 	THREE.Mesh.call(this);
 10 	this.blockHeight = (type.height ? type.height : CONFIG.BLOCK_HEIGHT)
 11 	this.geometry = new THREE.CubeGeometry(CONFIG.BLOCK_SIZE,this.blockHeight,CONFIG.BLOCK_SIZE);
 12 	this.position = new THREE.Vector3(           
 13 		(pos.x ? pos.x : 0),
 14 		-1*CONFIG.BLOCK_HEIGHT/2,
 15 		(pos.z ? pos.z : 0)
 16 	);                                        
 17 	this.material = new THREE.MeshLambertMaterial({color: (type.color ? type.color : 0x000000)});      
 18 	this.castShadow		= false;
 19 	this.receiveShadow	= true;
 20 	this.blockType = type;                
 21     this.left = this.position.x-(CONFIG.BLOCK_SIZE/2);
 22 	this.right = this.left+CONFIG.BLOCK_SIZE;
 23 	this.front = this.position.z-(CONFIG.BLOCK_SIZE/2);
 24 	this.back = this.front+CONFIG.BLOCK_SIZE;
 25 	this.neighbours = {};
 26 }
 27 
 28 Block.prototype = new THREE.Mesh();
 29 Block.prototype.constructor = Block;

 36 Block.prototype.updateNeighbours = function() {
 37                                                       
 38 	var col = parseInt(this.position.x/CONFIG.BLOCK_SIZE, 10);
 39 	var row = parseInt(this.position.z/CONFIG.BLOCK_SIZE, 10);
 40 	this.neighbours = {                                    
 41  	    frontLeft: game.track.blockForPosition(col-1, row-1),
 42 		frontRight: game.track.blockForPosition(col+1, row-1),
 43 		backLeft: game.track.blockForPosition(col-1, row+1),
 44 		backRight: game.track.blockForPosition(col+1, row+1),
 45     	front: game.track.blockForPosition(col, row-1),
 46 		left: game.track.blockForPosition(col-1, row),
 47 		right: game.track.blockForPosition(col+1, row),
 48 		back: game.track.blockForPosition(col, row+1)
 49 	};
 50 }

 52 /**
 53  * Returns an object of the collisions between the ball and this block
 54  * @param {THREE.Vector3} nextPos THREE.Vector3 object of the next track position
 55  * @returns String Array with the detected collisions (right|left|front|back)
 56  * @type Array
 57  * TODO: Make it better... ;-)
 58  */

 60 Block.prototype.getCollisions = function(nextPos) {
 61 
 62     var types = [];
 63     var ballPos = game.track.getBallPosition(nextPos);
 64     var withinWidth = (ballPos.x > this.left) && (ballPos.x < this.right);
 65     var withinLength = (ballPos.z > this.front) && (ballPos.z < this.back);
 66 
 67     var frontIntersection = (nextPos.z <= (this.back+game.ball.geometry.radius)*-1
 68         && withinWidth);
 69     var backIntersection = (nextPos.z >= (this.front-game.ball.geometry.radius)*-1
 70         && nextPos.z < this.back*-1
 71         && withinWidth);
 72     var rightIntersection = (nextPos.x <= (this.left-game.ball.geometry.radius)*-1
 73         && nextPos.x > (this.left + (CONFIG.BLOCK_SIZE/2)) * -1
 74         && withinLength);
 75     var leftIntersection = (nextPos.x >= (this.right+game.ball.geometry.radius)*-1
 76         && nextPos.x < (this.left + (CONFIG.BLOCK_SIZE/2)) * -1
 77         && withinLength);
 78 
 79     if (leftIntersection) {
 80 
 81         types.push("left");
 82     }
 83     else if (rightIntersection) {
 84 
 85         types.push("right");
 86     }
 87     if (frontIntersection) {
 88 
 89         types.push("front");
 90     }
 91     else if (backIntersection) {
 92 
 93         types.push("back");
 94     }
 95     return types;
 96 }

 98 /**
 99  * Finds the next track position, calculated by distance between this block and ball
100  * @param {THREE.Vector3} nextPos the track position in the next frame
101  * @param {String} type Type of collision (top|bottom|left|right)
102  * @returns Updated track position
103  * @type THREE.Vector3
104  */
105 
106 Block.prototype.getNextPositionToBall = function(nextPos, type) {
107 	
108 	if (!type && game.warpEndTime <= 0) {
109 		
110 		for (var dir in this.neighbours) {
111 			
112 			if (this.neighbours[dir]) {
113 				
114 				nextPos = this.neighbours[dir].getNextPositionToBall(nextPos, dir);
115 			}
116 		}
117 	}
118 	if (this.blockType.name == "blocker") {
119 		
120 		var floorIntersection = (game.track.position.y >= (-1*this.blockHeight/2));
121 		var collisionTypes = this.getCollisions(nextPos);
122 		if (floorIntersection) {
123 
124 			for (var i = 0; i < collisionTypes.length; i++) {
125 
126                 console.log(collisionTypes[i]);
127 				switch(collisionTypes[i]) {
128 
129 					case "right":
130 
131 						if (game.track.speedX >= 0) {
132 	
133 							nextPos.x = (this.left-game.ball.geometry.radius)*-1;
134 							game.track.speedX = 0;
135 							game.ball.canMove.right = false;                                             
136 							game.track.tempSpeedX = game.track.speedX;
137 						}
138 						break;    
139 
140 					case "left":
141 
142 						if (game.track.speedX <= 0) {
143 	
144 							nextPos.x = (this.right+game.ball.geometry.radius)*-1;
145 							game.track.speedX = 0;                                             
146 							game.ball.canMove.left = false;                                             
147 							game.track.tempSpeedX = game.track.speedX;
148 						}
149 						break;    
150 
151 					case "front":
152 
153 						if ((!type || type == "front") && game.track.speedZ >= 0) {
154 	
155 							nextPos.z = (this.back+game.ball.geometry.radius)*-1;
156 							game.track.lastZ = nextPos.z;
157 							game.track.speedZ = 0;
158 							game.track.tempSpeedZ = game.track.speedZ*game.timeRate;
159 							nextPos.z += game.track.tempSpeedZ;
160 							game.ball.canMove.front = false;
161 						}
162 						break;    
163 
164 					case "back":
165 
166 						if ((!type || type == "back") && game.track.speedZ <= 0) {
167 	
168 							nextPos.z = (this.front-game.ball.geometry.radius)*-1;
169 							game.track.speedZ = 0;                                             
170 							game.ball.canMove.back = false;
171 						}
172 						break;
173 				}		
174 			}
175 		}
176 	}
177 	                                   
178 	return nextPos;
179 }
